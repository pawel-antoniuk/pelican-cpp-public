[LAYOUT<{ELEMENT TYPE /element type, !PIXEL TYPE /pixel type}>] PLC LAYOUT:
	[SECTION<{/element type}>] HEADER
	[VECTOR(HEADER: NUMBER OF ELEMENTS)] ELEMENTS:
		[TYPE<ELEMENT TYPE>]: DATA<!/pixel type>

[SECTION<{ELEMENT TYPE}>] HEADER:
	"PLC\0" [4B](str)										- magic word
	[4B](uint) NUMBER OF ELEMENTS							- the number of elements
	[VECTOR<ELEMENT TYPE>(NUMBER OF ELEMENTS)] METADATAS:	- metadata for each element in file
		[ELEMENT TYPE: ID][1B](uint)						- type of element (see "IDS OF ELEMENTS" CONSTANTS)
		[4B](uint)											- offset of data of the element relative to end of "HEADER" SECTION

[TYPE] ELEMENT TYPE:
	TEXTURE 2D:
		[SECTION] ID:
			1 [1B][uint]
		[SECTION<PIXEL TYPE>] DATA:
			[PIXEL TYPE: ID]								- type of pixel
			[4B](uint) WIDTH								- width of texture
			[4B](uint) HEIGHT								- height of texture
			[VECTOR(WIDTH)] ROWS:
				[VECTOR(HEIGHT)] COLUMNS: 
					[PIXEL TYPE: DATA]
